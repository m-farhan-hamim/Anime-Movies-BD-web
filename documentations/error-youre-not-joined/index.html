async def handle_check_join(request): try: data = await request.json() user_id = data.get("user_id") if not user_id: return web.json_response({"joined": False}) # Check against Telegram API is_joined = await is_subscribed(user_id) return web.json_response({"joined": is_joined}) except Exception as e: print(f"Error checking join status: {e}") return web.json_response({"joined": False}) async def handle_inform(request): try: data = await request.json() message_text = f"📢 **New Anime Request!**\n\nSomeone couldn't find: `{data.get('movie_name', 'Unknown')}`\n\nAdmins, please upload it soon! 🚀" await call_telegram('sendMessage', {'chat_id': REQUEST_GROUP_ID, 'text': message_text, 'parse_mode': 'Markdown'}) return web.json_response({"status": "success"}) except: return web.json_response({"status": "error"}, status=500) async def handle_report(request): try: data = await request.json() # Get BD Time (+6) bd_tz = timezone(timedelta(hours=6)) current_time = datetime.now(bd_tz).strftime("%H:%M | %d/%m/%Y") report_text = ( f"🚨 **New User Report**\n\n" f"👤 **User:** {data.get('name')} ({data.get('username')})\n" f"⏰ **Time:** `{current_time}`\n\n" f"📝 **Reason/Issue:**\n`{data.get('issue')}`" ) await call_telegram('sendMessage', {'chat_id': REPORT_GROUP_ID, 'text': report_text, 'parse_mode': 'Markdown'}) return web.json_response({"status": "success"}) except: return web.json_response({"status": "error"}, status=500) async def send_main_menu(chat_id): reply_markup = {"inline_keyboard": [[{"text": "🎬 Open Anime Movies BD", "web_app": {"url": WEBAPP_URL}}]]} await call_telegram('sendMessage', { 'chat_id': chat_id, 'text': "Welcome to **Anime Movies BD**! 🍿\n\nClick the button below to search and watch your favorite anime.", 'parse_mode': 'Markdown', 'reply_markup': reply_markup }) async def handle_webhook(request): try: update = await request.json() if "callback_query" in update: call_id = update["callback_query"]["id"] user_id = update["callback_query"]["from"]["id"] chat_id = update["callback_query"]["message"]["chat"]["id"] data = update["callback_query"]["data"] if data == "verify_join": if await is_subscribed(user_id): await call_telegram('answerCallbackQuery', {'callback_query_id': call_id, 'text': "✅ Verified! Thanks for joining."}) await call_telegram('deleteMessage', {'chat_id': chat_id, 'message_id': update["callback_query"]["message"]["message_id"]}) await send_main_menu(chat_id) else: await call_telegram('answerCallbackQuery', {'callback_query_id': call_id, 'text': "❌ You haven't joined the channel yet!", 'show_alert': True}) return web.Response(text="OK") if "message" in update and "text" in update["message"]: text = update["message"]["text"] chat_id = update["message"]["chat"]["id"] user_id = update["message"]["from"]["id"] if text.startswith("/start"): if await is_subscribed(user_id): await send_main_menu(chat_id) else: reply_markup = { "inline_keyboard": [ [{"text": "📢 Join Channel", "url": CHANNEL_LINK}], [{"text": "✅ Verify", "callback_data": "verify_join"}] ] } await call_telegram('sendMessage', { 'chat_id': chat_id, 'text': "👋 **Hold up!**\n\nYou must join our official channel to use this bot and watch movies.", 'parse_mode': 'Markdown', 'reply_markup': reply_markup }) return web.Response(text="OK") except Exception as e: print(f"Webhook error: {e}") return web.Response(text="Error", status=500) async def handle_set_webhook(request): result = await call_telegram('setWebhook', {'url': f"{WEBAPP_URL}/webhook"}) if result.get('ok'): return web.Response(text="✅ Webhook successfully set! Bot is alive.") return web.Response(text=f"❌ Failed to set webhook: {result}") def init_app(): app = web.Application() app.router.add_get('/', handle_index) app.router.add_post('/check_join', handle_check_join) # 🆕 Added Endpoint app.router.add_post('/inform', handle_inform) app.router.add_post('/report', handle_report) app.router.add_post('/webhook', handle_webhook) app.router.add_get('/set_webhook', handle_set_webhook) return app if __name__ == '__main__': port = int(os.environ.get("PORT", 8080)) app = init_app() web.run_app(app, host='0.0.0.0', port=port)
